## ─────────────────────────────────────────────
## 1. CONTEXT
## ─────────────────────────────────────────────

Refactor the existing gemini.service.js into a unified llm.service.js
that supports multiple free AI models. Each wizard step can be configured
to use a specific model via environment variables.

This allows the team to:
- Switch models per step without changing code
- Use cheap/fast models for validation, powerful models for code generation
- Fall back to alternative models when quota is exhausted
- Test different models for quality comparison

Do NOT modify any existing routes. Only refactor the service layer.

## ─────────────────────────────────────────────
## 5. LLM SERVICE (llm.service.js)
## ─────────────────────────────────────────────

Main unified service:

```javascript
const { callGemini } = require('./providers/gemini.provider')
const { callGroq } = require('./providers/groq.provider')
const { callTogether } = require('./providers/together.provider')
const { callOpenRouter } = require('./providers/openrouter.provider')

// Parse model string "provider:model_name" → { provider, model }
function parseModelString(modelString) {
  if (!modelString) return { provider: 'gemini', model: 'gemini-2.5-flash' }
  const [provider, ...modelParts] = modelString.split(':')
  return { provider, model: modelParts.join(':') }
}

// Get model config for a specific step
function getStepModel(step) {
  const envMap = {
    'step_1_gdd':       process.env.STEP_1_GDD_MODEL,
    'step_2_sprites':   process.env.STEP_2_SPRITES_MODEL,
    'step_3_levels':    process.env.STEP_3_LEVELS_MODEL,
    'step_4_code':      process.env.STEP_4_CODE_MODEL,
    'step_5_audio':     process.env.STEP_5_AUDIO_MODEL,
    'validation':       process.env.VALIDATION_MODEL,
  }
  return parseModelString(envMap[step] || process.env.DEFAULT_MODEL)
}

// Main unified call function
async function callLLM(systemPrompt, userMessage, options = {}) {
  const step = options.step || null
  const { provider, model } = step
    ? getStepModel(step)
    : parseModelString(options.model || process.env.DEFAULT_MODEL)

  const callOptions = {
    ...options,
    model,
    maxOutputTokens: options.maxOutputTokens || 8192,
    temperature: options.temperature || 0.8
  }

  console.log(`[LLM] Step: ${step || 'default'} | Provider: ${provider} | Model: ${model}`)

  switch (provider) {
    case 'groq':        return callGroq(systemPrompt, userMessage, callOptions)
    case 'together':    return callTogether(systemPrompt, userMessage, callOptions)
    case 'openrouter':  return callOpenRouter(systemPrompt, userMessage, callOptions)
    case 'gemini':
    default:            return callGemini(systemPrompt, userMessage, callOptions)
  }
}

module.exports = { callLLM, getStepModel, parseModelString }
```

## 6. GEMINI PROVIDER (providers/gemini.provider.js)
## ─────────────────────────────────────────────

Move existing gemini.service.js logic here.
Keep the same logic but accept model as parameter:

```javascript
const { GoogleGenerativeAI } = require('@google/generative-ai')

const genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY)

async function callGemini(systemPrompt, userMessage, options = {}) {
  const model = options.model || 'gemini-2.5-flash'
  const startTime = Date.now()

  const geminiModel = genAI.getGenerativeModel({
    model,
    systemInstruction: systemPrompt,
    generationConfig: {
      responseMimeType: 'application/json',
      temperature: options.temperature || 0.8,
      maxOutputTokens: options.maxOutputTokens || 8192
    }
  })

  const result = await geminiModel.generateContent(userMessage)
  const response = result.response
  const text = response.text()

  // Clean markdown if present
  const cleaned = text
    .replace(/^```json\s*/i, '')
    .replace(/^```\s*/i, '')
    .replace(/```\s*$/i, '')
    .trim()

  const parsed = JSON.parse(cleaned)
  const usage = response.usageMetadata || {}

  return {
    data: parsed,
    meta: {
      provider: 'gemini',
      model,
      tokens_used: {
        input: usage.promptTokenCount || 0,
        output: usage.candidatesTokenCount || 0,
        cached: usage.cachedContentTokenCount || 0
      },
      duration_ms: Date.now() - startTime
    }
  }
}

module.exports = { callGemini }
```

## 7. GROQ PROVIDER (providers/groq.provider.js)
## ─────────────────────────────────────────────

Groq is extremely fast (tokens/second much faster than Gemini).
Uses OpenAI-compatible API. Install: npm install groq-sdk

```javascript
const Groq = require('groq-sdk')

const groq = new Groq({ apiKey: process.env.GROQ_API_KEY })

async function callGroq(systemPrompt, userMessage, options = {}) {
  const model = options.model || 'llama-3.3-70b-versatile'
  const startTime = Date.now()

  const response = await groq.chat.completions.create({
    model,
    temperature: options.temperature || 0.8,
    max_tokens: options.maxOutputTokens || 8192,
    response_format: { type: 'json_object' },
    messages: [
      { role: 'system', content: systemPrompt },
      { role: 'user', content: userMessage }
    ]
  })

  const text = response.choices[0].message.content
  const cleaned = text
    .replace(/^```json\s*/i, '')
    .replace(/^```\s*/i, '')
    .replace(/```\s*$/i, '')
    .trim()

  const parsed = JSON.parse(cleaned)

  return {
    data: parsed,
    meta: {
      provider: 'groq',
      model,
      tokens_used: {
        input: response.usage?.prompt_tokens || 0,
        output: response.usage?.completion_tokens || 0,
        cached: 0
      },
      duration_ms: Date.now() - startTime
    }
  }
}

module.exports = { callGroq }
```

## 13. ERROR HANDLING PER PROVIDER
## ─────────────────────────────────────────────

Each provider should handle its specific errors and map them
to a consistent error format:

{
  provider: 'groq',
  code: 'RATE_LIMIT' | 'INVALID_KEY' | 'MODEL_UNAVAILABLE' | 'PARSE_ERROR',
  message: 'human readable message',
  retry_after_ms: 60000 (if rate limited)
}

Specific error codes to catch:
- 429 → RATE_LIMIT, include retry_after from response headers
- 401 → INVALID_KEY
- 503 → MODEL_UNAVAILABLE
- JSON.parse failure → PARSE_ERROR

In gdd.routes.js update error handling:
  } catch (err) {
    const code = err.code || 'LLM_ERROR'
    const isRateLimit = err.status === 429 || code === 'RATE_LIMIT'
    return res.status(502).json({
      success: false,
      error: isRateLimit ? 'Rate limit reached. Try again later or switch models.' : 'LLM API call failed',
      code: isRateLimit ? 'RATE_LIMIT' : 'LLM_ERROR',
      retry_after_ms: err.retry_after_ms || null,
      ...(process.env.NODE_ENV === 'development' && { details: { message: err.message } })
    })
  }
